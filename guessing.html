const SPREADSHEET_ID = '1ryJDHmWyttWLcqV6A-jYxuiAm5k27yY2WGjVtOiTOJ8'; 

function doGet() {
  return HtmlService.createHtmlOutputFromFile('Index')
      .setTitle('Number Guessing Tournament')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

function saveScore(playerName, guesses, timeUsed) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheets()[2]; 
    sheet.appendRow([playerName, parseInt(guesses), parseInt(timeUsed), new Date()]);
    return true;
  } catch(e) {
    Logger.log("Error saveScore: " + e.toString());
    return false;
  }
}

function getLeaderboard() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheets()[2];
    const data = sheet.getDataRange().getValues();
    if (data.length <= 1) return [];
    
    const scores = [];
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      if (row[0] && !isNaN(parseInt(row[1])) && !isNaN(parseInt(row[2]))) {
        scores.push({
          name: row[0].toString(),
          clicks: parseInt(row[1]),
          timeUsed: parseInt(row[2])
        });
      }
    }
    
    scores.sort((a, b) => {
      if (a.clicks === b.clicks) return a.timeUsed - b.timeUsed;
      return a.clicks - b.clicks;
    });
    
    return scores.slice(0, 10);
  } catch(e) {
    Logger.log("Error getLeaderboard: " + e.toString());
    return [];
  }
}
